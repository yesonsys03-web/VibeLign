- [ ] **Step 3: SIM 구획 — 상태·추론 (spec §4 그대로)**

```js
// ===== SIM: 실제 guide.ts inferStep의 교육용 미러 =====
const initialSim = {
  hasPlanDoc: false,
  hasCheckpoint: false,
  changedFiles: [],   // [{ path, rev }] — rev = 실제 지문(path+status+mtime+size)의 단순화
  baseline: [],       // 체크포인트 시점 스냅샷
  guardOk: false,
  page: "home",
};

function countChanges(baseline, current) {
  // 실제 countChangesSinceBaseline의 미러: 경로 대칭차 + 같은 경로 rev 변화도 변경
  const b = new Map(baseline.map((f) => [f.path, f.rev]));
  const c = new Map(current.map((f) => [f.path, f.rev]));
  let n = 0;
  for (const [p, rev] of c) if (b.get(p) !== rev) n += 1;
  for (const p of b.keys()) if (!c.has(p)) n += 1;
  return n;
}

function inferStep(s) {
  // 실제 guide.ts inferStep과 동일 규칙(신호 이름도 동일 — 코드가 곧 교보재)
  let step = !s.hasPlanDoc ? 2 : !s.hasCheckpoint ? 3 : 4;
  if (s.hasCheckpoint && countChanges(s.baseline, s.changedFiles) > 0) {
    step = s.guardOk ? 6 : 5;
  }
  return step;
}
```
액션(useReducer): `confirmPlan`(hasPlanDoc=true) / `saveCheckpoint`(hasCheckpoint=true, baseline=changedFiles 스냅샷, guardOk=false) / `simAiEdit`(changedFiles에 `{path:"src/pages/Home.tsx",rev:1}`·`{path:"src/lib/api.ts",rev:1}` 추가 — 이미 있으면 무시) / `simReEdit`(changedFiles[0].rev += 1, **guardOk=false** ← 지문 리셋 미러) / `runGuard`(guardOk=true) / `setPage` / `reset`.
